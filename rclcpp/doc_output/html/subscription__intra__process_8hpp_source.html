146  if (any_callback_.use_take_shared_method()) {
147  ConstMessageSharedPtr shared_msg = shared_ptr->first;
148  any_callback_.dispatch_intra_process(shared_msg, msg_info);
149  } else {
150  MessageUniquePtr unique_msg = std::move(shared_ptr->second);
151  any_callback_.dispatch_intra_process(std::move(unique_msg), msg_info);
152  }
153  shared_ptr.reset();
154  }
